Application : Finding a book in a library
#include <stdio.h>
#include <string.h>

typedef struct {
    char title[50];
    char author[30];
} Book;

int binarySearch(Book books[], int n, char target[]) {
    int low = 0, high = n - 1;
    while (low <= high) {
        int mid = (low + high) / 2;
        int cmp = strcmp(books[mid].title, target);
        if (cmp == 0)
            return mid;
        else if (cmp < 0)
            low = mid + 1;
        else
            high = mid - 1;
    }
    return -1;
}

int main() {
    Book library[] = {
        {"Wings of Fire", "Dr.A.P.J Abdul Kalam"},
        {"Rich Dad Poor Dad", "Robert T.Kiyosaki"},
        {"The Alchemist", "Paulo Koelho"},
        {"Who moved my cheese", "Spencer Johnson"},
        {"The power of positive thinking", "Norman Vincent Peale"}
    };
    int n = sizeof(library) / sizeof(library[0]);
    char title[50];

    printf("Enter book title: ");
    fgets(title, sizeof(title), stdin);
    title[strcspn(title, "\n")] = 0;

    int index = binarySearch(library, n, title);

    if (index != -1)
        printf("Found: %s by %s\n", library[index].title, library[index].author);
    else
        printf("Book not found.\n");

    return 0;
}
