#include <stdio.h>
#include <stdlib.h>

// Function to swap two integers
void swap(int *a, int *b) {
    int t = *a;
    *a = *b;
    *b = t;
}

// Function to heapify (min-heap)
void heapify(int arr[], int n, int i) {
    int smallest = i;
    int left = 2*i + 1;
    int right = 2*i + 2;

    if (left < n && arr[left] < arr[smallest])
        smallest = left;

    if (right < n && arr[right] < arr[smallest])
        smallest = right;

    if (smallest != i) {
        swap(&arr[i], &arr[smallest]);
        heapify(arr, n, smallest);
    }
}

// Function to build min heap
void buildMinHeap(int arr[], int n) {
    for (int i = n / 2 - 1; i >= 0; i--)
        heapify(arr, n, i);
}

// Function to extract minimum from heap
int extractMin(int arr[], int *n) {
    int min = arr[0];
    arr[0] = arr[*n - 1];
    (*n)--;
    heapify(arr, *n, 0);
    return min;
}

// Function to insert a value into min heap
void insertHeap(int arr[], int *n, int value) {
    int i = (*n)++;
    arr[i] = value;
    while (i != 0 && arr[(i - 1) / 2] > arr[i]) {
        swap(&arr[i], &arr[(i - 1) / 2]);
        i = (i - 1) / 2;
    }
}

// Function to find optimal merge cost using greedy method
int optimalMerge(int files[], int n) {
    buildMinHeap(files, n);
    int totalCost = 0;

    while (n > 1) {
        int first = extractMin(files, &n);
        int second = extractMin(files, &n);
        int mergeCost = first + second;
        totalCost += mergeCost;
        insertHeap(files, &n, mergeCost);
    }

    return totalCost;
}

int main() {
    int n,i;
    printf("Enter number of files: ");
    scanf("%d", &n);

    int files[50];
    printf("Enter sizes of files: ");
    for (i = 0; i < n; i++)
        scanf("%d", &files[i]);

    int minCost = optimalMerge(files, n);
    printf("\nMinimum total cost of merging = %d\n", minCost);

    return 0;
}
